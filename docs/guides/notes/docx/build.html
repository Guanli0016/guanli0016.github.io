构建数据 ​

javascript
    // 创建文档 option数据格式见下方
    const doc = new docx.Document(option);

    // 打包文档 转成Blob数据
    docx.Packer.toBlob(doc).then(blob => {
        console.log(blob);
    });

    // 利用FileSaver里的saveAs方法 保存为docx格式的文件
    saveAs(blob, 'example.docx');
